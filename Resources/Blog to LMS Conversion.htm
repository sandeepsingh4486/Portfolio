            score++;

function sendCertificate() {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Form Responses 1"); // Change if needed
    var lastRow = sheet.getLastRow();
    var data = sheet.getRange(lastRow, 1, 1, sheet.getLastColumn()).getValues();

    var email = data[0][1]; // Assuming Email is in Column B
    var name = data[0][0]; // Assuming Name is in Column A
    var score = data[0][5]; // Assuming Score is in Column F

    if (score < 80) { // Only send certificates for scores 80%+
        return;
    }

    var doc = DocumentApp.create("Certificate - " + name);
    var body = doc.getBody();
    body.appendParagraph("Certificate of Completion")
        .setHeading(DocumentApp.ParagraphHeading.TITLE);
    body.appendParagraph("This is to certify that " + name + " has successfully completed the MATLAB LMS Quiz with a score of " + score + ".");
    body.appendParagraph("\n\nDate: " + new Date().toDateString());

    var pdf = doc.getAs('application/pdf');
    var file = DriveApp.createFile(pdf);

    MailApp.sendEmail({
        to: email,
        subject: "Your MATLAB LMS Certificate",
        body: "Congratulations " + name + "!\n\nYou have successfully passed the quiz. Please find your certificate attached.\n\nBest regards,\nMATLAB LMS Team",
        attachments: [file]
    });

    doc.setTrashed(true); // Delete doc after sending
}
